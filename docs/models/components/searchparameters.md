# SearchParameters

Common parameters for resource search operations.

## Example Usage

```typescript
import { SearchParameters } from "@cloudinary/assets/models/components";

let value: SearchParameters = {
  expression: "resource_type:image AND tags:kitten",
  sortBy: [
    "created_at",
    "public_id desc",
  ],
  maxResults: 30,
  aggregate: [
    "format",
    "resource_type",
  ],
  fields: "tags,context,metadata",
};
```
